import React from 'react';
import { connect } from 'react-redux';
import CalculatorDisplay from './CalculatorDisplay';
import CalculatorButtons from './CalculatorButtons';
import CalculatorMenu from './CalculatorMenu';

const Calculator = ({ uiData }) => {
  return (
    <div>
      <CalculatorDisplay displayValue={uiData.displayValue} />
      <CalculatorButtons buttons={uiData.buttons} />
      <CalculatorMenu menu={uiData.menu} />
    </div>
  );
};

const mapStateToProps = (state) => {
  return { uiData: state.calculator };
};

export default connect(mapStateToProps)(Calculator);

CalculatorButtons.js
```javascript
import React from 'react';
import { connect } from 'react-redux';

const CalculatorButtons = ({ buttons, onClickButton }) => {
  return (
    <div>
      {buttons.map((button) => (
        <button key={button.value} onClick={() => onClickButton(button.value)}>
          {button.label}
        </button>
      ))}
    </div>
  );
};

const mapDispatchToProps = (dispatch) => {
  return { onClickButton: (value) => dispatch({ type: 'BUTTON_CLICK', value }) };
};

export default connect(null, mapDispatchToProps)(CalculatorButtons);
```

calculatorReducer.js
```javascript
const initialState = {
  displayValue: '0',
  buttons: [],
  menu: []
};

const calculatorReducer = (state = initialState, action) => {
  switch (action.type) {
    case 'BUTTON_CLICK':
      return { ...state, displayValue: action.value };
    case 'MENU_CLICK':
      return { ...state, displayValue: '0' };
    case 'INITIALIZE_UI':
      return { ...state, ...action.uiData };
    default:
      return state;
  }
};

export default calculatorReducer;
```

store.js
```javascript
import { createStore, combineReducers } from 'redux';
import calculatorReducer from './calculatorReducer';

const rootReducer = combineReducers({
  calculator: calculatorReducer
});

const store = createStore(rootReducer);

export default store;
```

App.js
```javascript
import React, { useEffect } from 'react';
import { Provider } from 'react-redux';
import store from './store';
import Calculator from './Calculator';
import axios from 'axios';

const App = () => {
  useEffect(() => {
    axios.get('/api/initializeUI')
      .then(response => {
        store.dispatch({ type: 'INITIALIZE_UI', uiData: response.data });
      })
      .catch(error => {
        console.error(error);
      });
  }, []);

  return (
    <Provider store={store}>
      <Calculator />
    </Provider>
  );
};

export default App;
```
